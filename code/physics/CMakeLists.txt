nebula_begin_lib(physics)

target_include_directories(physics PUBLIC ${FIPS_DEPLOY_DIR}/physx/include ${CMAKE_CURRENT_SOURCE_DIR} ${CODE_ROOT}/physics)

fips_deps(foundation resource render nflatbuffer)
fips_files(physicsinterface.h physicsinterface.cc)
fips_dir(physics)
fips_files( 
            actorcontext.cc
            actorcontext.h
            streamactorpool.cc
            streamactorpool.h
            callbacks.cc
            callbacks.h
            debugui.cc
            debugui.h
            utils.h
            visualdebugger.cc
            visualdebugger.h
            physxstate.cc
            physxstate.h
            )
nebula_flatc(SYSTEM physics/actor.fbs physics/collisions.fbs physics/material.fbs)
nebula_end_lib()
